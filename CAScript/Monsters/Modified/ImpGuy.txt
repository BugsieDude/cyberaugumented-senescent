Class DCY_Impe : DCYMonster
{
  Default
  {
  Tag "$FN_IMP";
  Health 60;
  Radius 20;
  Height 56;
  Mass 100;
  Speed 8;
  PainChance 200;
  SeeSound "imp/sight";
  PainSound "imp/pain";
  DeathSound "imp/death";
  ActiveSound "imp/active";
  HitObituary "$OB_IMPHIT";
  Obituary "$OB_IMP";
  Species "Imps";
  }
  
  void A_ImpLunge()
  {
	A_FaceTarget();
	A_Recoil(-10);
	A_ChangeVelocity(0, 0, 1.0, CVF_RELATIVE);
  }
  
  States
  {
  Spawn:
	TROO A 10 DCY_Look();
	Loop;
  Look:
	TROO AABBCCDD 4
	{
		DCY_LookingForPlayer();
		
		if (random(0, 50) == 1)
			SetStateLabel("Stand");
	}
	Loop;
  Stand:
	TROO AAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAA 1 A_Chase(null, "Alert", CHF_NOPLAYACTIVE|CHF_DONTMOVE);
	Goto Look;
  Alert:
	"####" "#" 0 DCY_AlertSound();
	Goto See;
  See:
	TROO AABBCCDD 3 A_Chase();
	Loop;
  Melee:
	TROO E 0;
	TROO EF 5 A_FaceTarget();
	TROO G 5
	{
		A_CustomMeleeAttack(3 * random(1, 8), "imp/melee", "");
		A_FaceTarget();
	}
	TROO F 5;
	Goto See;
  SLASHER:
	TROO E 1 A_ImpLunge();
	TROO EFGF 4 A_FaceTarget();
	TROO E 1 A_ImpLunge();
	TROO F 5 A_FaceTarget();
	TROO G 5 A_SpawnProjectile("DCY_ImpFireball", 32, 0, frandom(-0.5, 0.5));
	TROO F 3;
	Goto See;
  Missile:
	TROO E 0 A_JumpIfCloser(128, "SLASHER");
	TROO E 0 A_Jump(100, "Dodge", "Run");
	TROO EF 8 A_FaceTarget();
	TROO G 4
	{
		A_SpawnProjectile("DCY_ImpFireball", 32, 0, frandom(-0.5, 0.5));
		A_FaceTarget();
	}
	TROO F 4;
	Goto See;
  Dodge:
	TROO A 15 A_ChangeVelocity(randompick(-10, 10), 0, 1.0, CVF_RELATIVE);
	TROO A 1 A_Stop();
	Goto Missile+2;
  Run:
	TROO ABCD 3
	{
		for (int i = 2; i > 0; i--)
			A_Chase(null, null);
	}
	TROO EF 8 A_FaceTarget();
	TROO G 4
	{
		A_SpawnProjectile("DCY_ImpFireball", 32, 0, frandom(-0.5, 0.5));
		A_FaceTarget();
	}
	TROO F 4;
	TROO ABCD 3
	{
		for (int i = 2; i > 0; i--)
			A_Chase(null, null);
	}
	TROO EF 8 A_FaceTarget();
	TROO G 4
	{
		A_SpawnProjectile("DCY_ImpFireball", 32, 0, frandom(-0.5, 0.5));
		A_ChangeVelocity(randompick(-10, 10), 0, 1.0, CVF_RELATIVE);
		A_FaceTarget();
	}
	TROO F 4;
	TROO F 1 A_Stop();
	Goto See;
  Pain:
	TROO H 4;
	TROO H 4 A_Pain();
	Goto See;
  Death:
    TNT1 A 0 A_Jump(128, "Death2");
	Goto NormalDeath;
  NormalDeath:
    TROO I 1 A_Flip();
	TROO I 1 A_Scream();
	TROO I 5 A_NoBlocking();
	TROO JKL 4;
	TROO M -1;
	Stop;
  Death2:
    1D_3 A 1;
	1D_3 A 6 A_ScreamAndUnblock();
	1D_3 CD 7;
	1D_3 E -1;
	Stop;
  XDeath:
    TNT1 A 0 A_Jump(128, "XDeath1");
	Goto XDeathNormal;
  XDeathNormal:
    TROO N 1;
	TROO N 1 DCY_Gibs();
	TROO N 5 A_NoBlocking();
	TROO OPQRST 4;
	TROO U -1;
	Stop;
  XDeath1:
    1D_1 A 6 A_Scream();
	1D_1 A 0 A_StopSound(CHAN_VOICE);
	1D_1 B 6 DCY_Gibs();
	TNT1 A 0 A_NoBlocking();
	1D_1 CDEFGH 5;
	1D_1 I -1;
	Stop;
  Raise:
	TROO M 3;
    TROO LKHJI 4;
	TROO H 4 A_Restore();
    Goto See;
  }
}
