Class DCY_Backpack : Backpack Replaces Backpack
{
  Default
  {
  Inventory.PickupSound "dcy/backpack";
  Inventory.PickupMessage "Time to get those weapons locked up. (Ammo increased.)";
  }
  
  States
  {
  Spawn:
    BCPA ABCDEFG 3;
    Loop;
  }
}

Class DCY_Clip : Clip Replaces Clip
{
  Default
  {
  Inventory.PickupSound "dcy/clip";
  Inventory.PickupMessage "Picked up a clip of bullets.";
  }
  
  States
  {
  Spawn:
	DCA_ A -1;
	Loop;
  Pickup:
	DCA_ A 0
	{
		A_TakeInventory("DCY_Clip");
		A_GiveInventory("Clip");
	}
	Stop;
  }
}
